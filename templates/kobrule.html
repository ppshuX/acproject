    static class Cell {//è›‡èº«ä½“ï¼ˆå•æ ¼ï¼‰

    private boolean check_tail_increasing(int step) {//æ£€æŸ¥è›‡ä»€ä¹ˆæ—¶å€™ä¼šå˜é•¿

    public List<Cell> getCells(int sx, int sy, String steps) {//è·å–æ¸¸æˆä¸­ä¸¤æ¡è›‡çš„èº«ä½“ä½ç½®
        steps=steps.substring(1,steps.length()-1);

    }
